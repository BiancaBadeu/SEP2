drop schema sep cascade;
create schema sep;
set schema 'sep';

create domain v20 as varchar(20);
create domain v50 as varchar(50);
create domain v1000 as varchar(1000);

DROP TABLE IF EXISTS movies;
CREATE TABLE movies (

    movieName v50 PRIMARY KEY ,
    director v20 not null ,
    length integer not null ,
    movieDescription v1000 not null ,
    avgRating decimal(2,1) not null,
    releaseYear INTEGER Not Null,
    genre v50 not null
);

DROP TABLE IF EXISTS Person;
CREATE TABLE Person (
  name VARCHAR(60) NOT NULL ,
  phoneNumber VARCHAR(40) NOT NULL ,
  userName VARCHAR(40) PRIMARY KEY ,
  password VARCHAR(40) NOT NULL,
  age integer not null
);

DROP TABLE IF EXISTS reviewsAndRatings;
CREATE TABLE reviewsAndRatings(

    movieTitle v50 not null,
    reviewID serial,
    rating integer NOT NULL,
    comment varchar(500) NOT NULL,
    userName varchar(40) NOT NULL ,

    FOREIGN KEY (movieTitle) references movies(movieName),
    FOREIGN KEY (userName) references Person(userName),
    PRIMARY KEY (movieTitle, userName,reviewID)
);

DROP TABLE IF EXISTS rentals;
create table rentals(
    expirationDate date not null ,
    userName varchar(40) not null ,
    movieTitle v50 not null ,

    FOREIGN KEY (userName) references Person(userName),
    foreign key (movieTitle) references movies(movieName),
    PRIMARY KEY (userName, movieTitle)
);
